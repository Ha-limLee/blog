# Objective

# Use case

저번 action 코드를 보자

```js
// @ts-check
import { createAction } from "redux-actions";

/**
 * @typedef {import("./api").Todo} Todo
 */

export const SET_TODO = 'Todo/ADD_TODO';
export const setTodo = /** @type {typeof createAction<Todo>} */ (createAction)(SET_TODO);

```
